下面给你一套**可直接落地的 tinyRtc SDK 目录结构 + 头文件/API 设计稿**。目标是：**跨平台（POSIX/RTOS/小程序适配层）+ 清晰分层 + 稳定对外接口**。

---

# 一、SDK 总体目录结构

```text
tinyrtc/
├── include/                    # 对外头文件（稳定API）
│   ├── trtc.h                 # 总入口（只暴露必要接口）
│   ├── trtc_types.h
│   ├── trtc_error.h
│   ├── trtc_config.h
│   │
│   ├── platform/
│   │   └── trtc_platform.h    # 平台抽象接口
│   │
│   ├── core/
│   │   ├── trtc_socket.h
│   │   ├── trtc_thread.h
│   │   ├── trtc_mem.h
│   │   ├── trtc_time.h
│   │   └── trtc_log.h
│   │
│   └── rtc/
│       ├── trtc_rtc.h         # RTC主模块
│       ├── trtc_peer.h
│       ├── trtc_session.h
│       └── trtc_packet.h
│
├── src/                        # 实现
│   ├── core/
│   ├── rtc/
│   └── platform/
│       ├── posix/
│       ├── rtos/
│       └── miniapp/           # 小程序适配（你之前提到）
│
├── port/                       # 用户自定义移植层（可选）
│   └── trtc_port.c
│
├── examples/
├── tests/
└── CMakeLists.txt
```

# 二、对外统一入口（include/trtc.h）

👉 **只暴露高层API，不暴露底层实现**

```c
#ifndef TRTC_H
#define TRTC_H

#include "trtc_types.h"
#include "trtc_error.h"
#include "trtc_config.h"

#include "rtc/trtc_rtc.h"

#ifdef __cplusplus
extern "C" {
#endif

int trtc_init(void);
void trtc_deinit(void);

#ifdef __cplusplus
}
#endif

#endif
```

# 五、平台抽象层（核心关键）

## include/platform/trtc_platform.h

👉 所有系统依赖都必须走这里

```c
#ifndef TRTC_PLATFORM_H
#define TRTC_PLATFORM_H

#include <stddef.h>

#ifdef __cplusplus
extern "C" {
#endif

/* memory */
void *trtc_platform_malloc(size_t size);
void trtc_platform_free(void *ptr);

/* log */
void trtc_platform_log(const char *fmt, ...);

/* thread */
typedef void (*trtc_thread_func)(void *arg);

void *trtc_platform_thread_create(trtc_thread_func func, void *arg);
void trtc_platform_thread_sleep(int ms);

/* socket */
int trtc_platform_socket_create(void);
int trtc_platform_socket_connect(int sock, const char *ip, int port);
int trtc_platform_socket_send(int sock, const void *data, int len);
int trtc_platform_socket_recv(int sock, void *buf, int len);
void trtc_platform_socket_close(int sock);

#ifdef __cplusplus
}
#endif

#endif
```

# 七、RTC核心API（对外核心能力）

## include/rtc/trtc_rtc.h

```c
#ifndef TRTC_RTC_H
#define TRTC_RTC_H

#include "trtc_types.h"

#ifdef __cplusplus
extern "C" {
#endif

typedef struct trtc_rtc trtc_rtc_t;

/* 回调 */
typedef void (*trtc_on_data_cb)(const uint8_t *data, uint32_t len);
typedef void (*trtc_on_state_cb)(int state);

/* 配置 */
typedef struct {
    const char *stun_server;
    const char *turn_server;
} trtc_rtc_config_t;

/* 生命周期 */
trtc_rtc_t *trtc_rtc_create(const trtc_rtc_config_t *config);
void trtc_rtc_destroy(trtc_rtc_t *rtc);

/* 连接 */
int trtc_rtc_connect(trtc_rtc_t *rtc, const char *peer_id);

/* 数据 */
int trtc_rtc_send(trtc_rtc_t *rtc, const uint8_t *data, uint32_t len);

/* 回调注册 */
void trtc_rtc_set_on_data(trtc_rtc_t *rtc, trtc_on_data_cb cb);
void trtc_rtc_set_on_state(trtc_rtc_t *rtc, trtc_on_state_cb cb);

#ifdef __cplusplus
}
#endif

#endif
```

# 十、设计亮点（你这个SDK的优势）

### ✔ 完全可移植

* 所有系统调用 → platform层

### ✔ 清晰分层

* platform（系统）
* core（基础能力）
* rtc（业务）

# 十一、下一步可以做的优化

如果你要做“工程级SDK”，建议再加：

### 1️⃣ 统一上下文对象（避免全局变量）

```c
typedef struct trtc_context trtc_context_t;
```

---

### 2️⃣ 内存替换能力（IoT很需要）

```c
void trtc_set_mem_ops(...);
```

---

### 3️⃣ 日志重定向

```c
void trtc_set_log_callback(...);
```
